    solvent

    Simple dependency resolver designed for validating typed dependency contraints.

    Install

    npm i -S solvent

    Usage

    import solvent from 'solvent'

export default function (dependencies) {
  const { React, connect } = solvent( { React: 'object'
                                      , connect: 'function'
                                      } )(dependencies)

  /** DO SOMETHING */
}
